(use-package 'guide-key)
(setq guide-key/guide-key-sequence '("s-m" "C-x 4"))

(setq guide-key/guide-key-sequence '("C-x"))
(setq guide-key/recursive-key-sequence-flag t)

(defun guide-key/my-hook-function-for-org-mode ()
